CREATE TABLE `user` (
`id` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
`user` TEXT,
`pins` TEXT,
`idpins` INTEGER NOT NULL
);

CREATE TABLE `pins` (
`id` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
`url` TEXT,
FOREIGN KEY(id) REFERENCES user(idpins)
);

CREATE TABLE `comment` (
`id` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
`userid` INTEGER NOT NULL,
`comment` TEXT,
`idpins` INTEGER NOT NULL,
FOREIGN KEY(userid) REFERENCES user(id),
FOREIGN KEY(idpins) REFERENCES user(idpins)
);
