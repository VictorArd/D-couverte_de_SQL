CREATE TABLE cours (
`id` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
`name` TEXT
);

CREATE TABLE eleves (
`id` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
`eleve` TEXT,
`inscrit` INTEGER NOT NULL,
FOREIGN KEY(inscrit) REFERENCES cours(id)
);
